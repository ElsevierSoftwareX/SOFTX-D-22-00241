7class Matlab(object): 

8 r""" 

9 A bridge object that allows python to interact with a MATLAB session 

10 """ 

11 def __init__(self, kwave_path=None, offgrid_path=None): 

12 """Constructs a new Matlab object. 

13 

14 Args: 

15 kwave_path ([type], optional): Path to the k-Wave toolbox. 

16 If `None`, the path is loaded from the environment variable 

17 `KWAVE_CORE_PATH`. 

18 offgrid_path ([type], optional): Path to the Off-Grid-Sources toolbox. 

19 If `None`, the path is loaded from the environment variable 

20 `KWAVE_OFFGRID_PATH`. 

21 

22 [^1]: [http://www.k-wave.org/](http://www.k-wave.org/) 

23 

24 [^2]: Elliott S. Wise et al., 2019. [*Representing arbitrary acoustic source  

25 and sensor distributionsin Fourier collocation methods*](https://bug.medphys.ucl.ac.uk/papers/2019-Wise-JASA.pdf) 

26 

27 !!! example 

28 ```python 

29 from jwave.matlab import Matlab 

30 mlb = Matlab() 

31 ``` 

32 """ 

33 # Init variables 

34 self._engine = None 

35 if offgrid_path is None: 

36 self.kwave_path = os.environ.get('KWAVE_CORE_PATH') 

37 if offgrid_path is None: 

38 self.offgrid_path = os.environ.get('KWAVE_OFFGRID_PATH') 

39 

40 # Redirect outputs 

41 self.out = io.StringIO() 

42 self.err = io.StringIO() 

43 

44 def start(self, join_session=True): 

45 """Starts a MATLAB engine session. 

46 

47 Args: 

48 join_session (bool, optional): If True, it tries to connect to 

49 a running instance of MATLAB. If that fails or `joint_session=False`, 

50 starts a new MATLAB Session 

51 """ 

52 # If a shared matlab istance is running, use it 

53 open_matlabs = matlab.engine.find_matlab() 

54 if len(open_matlabs) > 0: 

55 print("Connecting to Matlab session {}".format(open_matlabs[0])) 

56 self._engine = matlab.engine.connect_matlab(open_matlabs[0]) 

57 else: 

58 print("Opening new matlab session") 

59 self._engine = matlab.engine.start_matlab() 

60 

61 # Add paths 

62 self._engine.addpath(self.kwave_path) 

63 self._engine.addpath(self.offgrid_path) 

64 

65 def run_script(self, script_path: str): 

66 """Runs a MATLAB script 

67 

68 Args: 

69 script_path (str): Script path. 

70 """ 

71 self._engine.eval("run('"+script_path+"');", nargout=0) 

72 

73 def run(self, command: str, to=None, nargout=0): 

74 """Runs a MATLAB command in the attached session. 

75 

76 Args: 

77 command (str): Command to be run. 

78 to (str, optional): If not `None`, saves the output of the command 

79 in the MATLAB workspace variable with the given string as name.  

80 nargout (int, optional): If not `0`, returns the outputs of the function. 

81 

82 Raises: 

83 RuntimeError: If the MATLAB engine is not running. Remember to 

84 start the MATLAB session using the `start()` method before 

85 executing MATLAB functions. 

86 

87 Returns: 

88 [Any]: Command output 

89 """ 

90 

91 if self._engine is None: 

92 raise RuntimeError("Matlab engine is not running, please use `start()` method") 

93 

94 if to is None: 

95 return self._engine.eval(command, nargout=nargout, stdout=self.out,stderr=self.err) 

96 else: 

97 self._engine.workspace[to] = self._engine.eval(command, stdout=self.out,stderr=self.err) 

98

101

106 if isinstance(value, list): # List transformed to matlab array

108 self._engine.workspace[variable_name]=value

113 Args:

116 Returns: 

117 [type]: [description]

119 return self._engine.workspace[variable_name]
